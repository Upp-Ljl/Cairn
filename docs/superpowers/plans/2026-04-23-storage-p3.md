# Cairn Storage P3 (tasks + sub_agents + scratchpad + lanes FK) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add the three remaining core tables (`tasks`, `sub_agents`, `scratchpad`) plus the MCP-wedge NULL-task assignment helper, and re-establish the FK constraints on `lanes` that P1 deliberately omitted, all via table-rewrite migration 006.

**Architecture:** Four additive migrations (003, 004, 005, 006) registered in order in `migrations/index.ts`. Migrations 003-005 are straightforward `CREATE TABLE` + indexes. Migration 006 is the canonical SQLite table-rewrite dance (`PRAGMA foreign_keys=OFF` → `BEGIN` → `CREATE TABLE lanes_new` with FKs → `INSERT SELECT` → `DROP` → `RENAME` → re-create indexes → `PRAGMA foreign_key_check` → `COMMIT` → `PRAGMA foreign_keys=ON`). Repositories for the three new tables wrap `better-sqlite3` with CAS-style status transitions (sub_agents starts/ends timestamps), 24h default TTL for scratchpad, and a startup GC hook extending P2's `recovery.ts`. Scratchpad never eagerly deletes on read — `getScratch` returns `null` for expired rows and leaves them for the GC pass (per OQ-5 option A). Per OQ-1 option C, NULL `task_id` lanes are retained; `lanes-mcp-wedge.ts` provides `assignLanesToNewTask` to let the UI bulk-assign them to a fresh task.

**Tech Stack:** TypeScript (strict, ESM with `.js` import suffixes), `better-sqlite3`, `vitest` (including `vi.setSystemTime` for TTL tests), `ulid`, Node 20+.

**Spec:** `DESIGN_STORAGE.md` §4.2 (tasks), §4.3 (sub_agents), §4.5 (lanes FKs), §4.8 (scratchpad), §9 Query #6, §15 OQ-1 (option C — keep NULL + helper), §15 OQ-5 (option A — required `expires_at`, default 24h, startup GC).

**Deviation from spec:** None intentional. Phase 4 concerns (`tasks.description_vec` population) are explicitly deferred; P3 only adds the column as NULL and does not read or write it. Periodic (beyond-startup) scratchpad GC is deferred to post-v0.1.

```ts
import type { Database } from 'better-sqlite3';
import { ulid } from 'ulid';
import type { TaskRow, TaskStatus } from '../types.js';

export interface NewTaskInput {
  title: string;
  description?: string | null;
}

export function createTask(db: Database, input: NewTaskInput): TaskRow {
  const now = Date.now();
  const id = ulid();
  db.prepare(
    `INSERT INTO tasks (id, title, description, description_vec, status, created_at, updated_at)
     VALUES (?, ?, ?, NULL, 'PENDING', ?, ?)`,
  ).run(id, input.title, input.description ?? null, now, now);
  return getTaskById(db, id)!;
}

export function getTaskById(db: Database, id: string): TaskRow | null {
  const row = db.prepare(`SELECT * FROM tasks WHERE id = ?`).get(id) as
    | TaskRow
    | undefined;
  return row ?? null;
}

export function listTasks(
  db: Database,
  filter: { status?: TaskStatus },
): TaskRow[] {
  if (filter.status !== undefined) {
    return db
      .prepare(
        `SELECT * FROM tasks WHERE status = ? ORDER BY updated_at DESC`,
      )
      .all(filter.status) as TaskRow[];
  }
  return db
    .prepare(`SELECT * FROM tasks ORDER BY updated_at DESC`)
    .all() as TaskRow[];
}

/**
 * CAS-style transition. If oldStatus is provided, only updates when the
 * current status matches. Returns true iff a row was updated. Always bumps
 * updated_at.
 */
export function updateTaskStatus(
  db: Database,
  id: string,
  newStatus: TaskStatus,
  oldStatus?: TaskStatus,
): boolean {
  const now = Date.now();
  const result =
    oldStatus === undefined
      ? db
          .prepare(
```

```ts
  name: string;
  role?: string | null;
}

export function createSubAgent(
  db: Database,
  taskId: string,
  input: NewSubAgentInput,
): SubAgentRow {

  const id = ulid();
  db.prepare(

       (id, task_id, name, role, status, started_at, ended_at, created_at)

export function getSubAgentById(
  db: Database,
  id: string,

    .get(id) as SubAgentRow | undefined;
  return row ?? null;

export function listSubAgentsByTask(
  db: Database,
  taskId: string,
  filter: { status?: SubAgentStatus },
): SubAgentRow[] {
  if (filter.status !== undefined) {
    return db
      .prepare(
        `SELECT * FROM sub_agents
         WHERE task_id = ? AND status = ?
         ORDER BY created_at ASC`,
      )
      .all(taskId, filter.status) as SubAgentRow[];
  }
  return db
    .prepare(
      `SELECT * FROM sub_agents WHERE task_id = ? ORDER BY created_at ASC`,
    )
    .all(taskId) as SubAgentRow[];
}

/**
 * Transition rules baked into the UPDATE:
 *  - If new status is RUNNING and started_at IS NULL, set started_at = now.
 *  - If new status is DONE or FAILED, set ended_at = now (always).
 *
 * Returns true iff a row was updated.
 */
export function updateSubAgentStatus(
  db: Database,
  id: string,
  newStatus: SubAgentStatus,
): boolean {
  const now = Date.now();
  let sql: string;
  if (newStatus === 'RUNNING') {
    sql = `UPDATE sub_agents
             SET status = ?,
                 started_at = COALESCE(started_at, ?)
           WHERE id = ?`;
    const r = db.prepare(sql).run(newStatus, now, id);
    return r.changes === 1;
  }
  if (newStatus === 'DONE' || newStatus === 'FAILED') {
    sql = `UPDATE sub_agents
             SET status = ?, ended_at = ?
           WHERE id = ?`;
    const r = db.prepare(sql).run(newStatus, now, id);
    return r.changes === 1;
  }
  sql = `UPDATE sub_agents SET status = ? WHERE id = ?`;
  const r = db.prepare(sql).run(newStatus, id);
  return r.changes === 1;
}
```

```ts
import type { Database } from 'better-sqlite3';
import * as fs from 'node:fs';
import * as path from 'node:path';
import * as crypto from 'node:crypto';
import type { ScratchRow } from '../types.js';

export const DEFAULT_SCRATCH_TTL_MS = 24 * 60 * 60 * 1000;
// Inline/external threshold reused from §2.1 (128 KB).
const INLINE_THRESHOLD_BYTES = 128 * 1024;

export interface PutScratchInput {
  key: string;
  value: unknown;
  task_id?: string | null;
  /** Defaults to DEFAULT_SCRATCH_TTL_MS. Pass a finite positive number. */
  ttl_ms?: number;
}

function writeBlobIfLarge(
  blobRoot: string,
  serialized: string,
): { value_json: string | null; value_path: string | null } {
  const byteLen = Buffer.byteLength(serialized, 'utf8');
  if (byteLen < INLINE_THRESHOLD_BYTES) {
    return { value_json: serialized, value_path: null };
  }
  const sha = crypto.createHash('sha256').update(serialized).digest('hex');
  const dir = path.join(blobRoot, sha.slice(0, 2));
  fs.mkdirSync(dir, { recursive: true });
  const full = path.join(dir, sha);
  if (!fs.existsSync(full)) {
    fs.writeFileSync(full, serialized, 'utf8');
  }
  return { value_json: null, value_path: full };
}

export function putScratch(
  db: Database,
  blobRoot: string,
  input: PutScratchInput,
): void {
  const now = Date.now();
  const ttl = input.ttl_ms ?? DEFAULT_SCRATCH_TTL_MS;
  const expiresAt = now + ttl;
  const serialized = JSON.stringify(input.value);
  const { value_json, value_path } = writeBlobIfLarge(blobRoot, serialized);

  db.prepare(
    `INSERT INTO scratchpad
       (key, value_json, value_path, task_id, expires_at, created_at, updated_at)
     VALUES (?, ?, ?, ?, ?, ?, ?)
     ON CONFLICT(key) DO UPDATE SET
       value_json = excluded.value_json,
       value_path = excluded.value_path,
       task_id    = excluded.task_id,
       expires_at = excluded.expires_at,
       updated_at = excluded.updated_at`,
  ).run(
    input.key,
    value_json,
    value_path,
    input.task_id ?? null,
    expiresAt,
    now,
    now,
  );
}

/**
 * Returns the deserialized value, or null if the key is missing OR expired.
 * Does NOT delete expired rows — gcExpiredScratch is responsible.
 */
export function getScratch(
  db: Database,
  _blobRoot: string,
  key: string,
): unknown {
  const row = db
    .prepare(
      `SELECT value_json, value_path, expires_at FROM scratchpad WHERE key = ?`,
    )
    .get(key) as
    | { value_json: string | null; value_path: string | null; expires_at: number | null }
    | undefined;
  if (!row) return null;
  if (row.expires_at !== null && row.expires_at <= Date.now()) {
    return null;
  }
  if (row.value_json !== null) {
    return JSON.parse(row.value_json);
  }
  if (row.value_path !== null) {
    const raw = fs.readFileSync(row.value_path, 'utf8');
    return JSON.parse(raw);
  }
  return null;
}

export function deleteScratch(db: Database, key: string): void {
  db.prepare(`DELETE FROM scratchpad WHERE key = ?`).run(key);
}

export function listScratchByTask(
  db: Database,
  taskId: string,
): ScratchRow[] {
  return db
    .prepare(
      `SELECT * FROM scratchpad WHERE task_id = ? ORDER BY updated_at DESC`,
    )
    .all(taskId) as ScratchRow[];
}

/**
 * Deletes all rows where expires_at IS NOT NULL AND expires_at <= now.
 * Returns the number of deleted rows.
 */
export function gcExpiredScratch(db: Database): number {
  const now = Date.now();
  const r = db
    .prepare(
      `DELETE FROM scratchpad WHERE expires_at IS NOT NULL AND expires_at <= ?`,
    )
    .run(now);
  return r.changes;
}
```

```ts
import type { Migration } from './runner.js';

/**
 * SQLite cannot ALTER TABLE ADD CONSTRAINT, so we use the canonical
 * table-rewrite dance to re-establish the foreign keys on `lanes` that
 * migration 001 deliberately omitted (see DESIGN_STORAGE.md §4.5).
 *
 * Ordering within up():
 *   1. foreign_keys OFF (so INSERT...SELECT backfill does not reject
 *      pre-existing dangling rows — per OQ-1, we keep NULL / stale task_id
 *      lanes instead of deleting them).
 *   2. Create lanes_new with the full FK set.
 *   3. Copy rows.
 *   4. Drop old, rename new.
 *   5. Re-create all 5 lane indexes enumerated in migration 001.
 *   6. PRAGMA foreign_key_check — if any row is reported, throw so the
 *      outer transaction in the runner rolls back.
 *   7. foreign_keys ON.
 *
 * The runner already wraps up() in a db.transaction(); inside that
 * transaction, toggling PRAGMA foreign_keys has no effect in SQLite (the
 * docs explicitly state this — the PRAGMA only takes effect outside a
 * transaction). better-sqlite3's runner used by Cairn here uses
 * IMMEDIATE-style transactions; to work around the pragma-in-transaction
 * limitation, this migration opens and closes its own internal handling:
 * we EXEC the entire rewrite as a single batch where the first and last
 * statements are the pragma toggles. Because CREATE/DROP TABLE auto-commits
 * in SQLite if outside an explicit transaction, but the runner wraps us in
 * one, we rely on a documented idiom: issue the PRAGMA before the runner's
 * BEGIN by exporting a preUp hook — in practice Cairn's runner is simple
 * enough that we can call the pragma toggles with a plain db.exec() here;
 * SQLite ignores the toggle while in a tx, and since FK enforcement already
 * defers to the end of the statement in better-sqlite3 during INSERT...SELECT,
 * the backfill completes cleanly regardless. The explicit foreign_key_check
 * at the end is what actually guards correctness.
 */
export const migration006LanesFk: Migration = {
  version: 6,
  name: 'lanes_fk',
  up: (db) => {
    // Attempt to disable FK enforcement for this migration. This is a no-op
    // while inside a transaction, but harmless; the foreign_key_check below
    // is the actual correctness guard.
    db.exec(`PRAGMA foreign_keys = OFF;`);

    db.exec(`
      CREATE TABLE lanes_new (
        id               TEXT    PRIMARY KEY,
        task_id          TEXT    REFERENCES tasks(id)       ON DELETE SET NULL,
        sub_agent_id     TEXT    REFERENCES sub_agents(id)  ON DELETE SET NULL,
        checkpoint_id    TEXT    REFERENCES checkpoints(id),
        endpoint         TEXT    NOT NULL,
        scenario         TEXT,
        state            TEXT    NOT NULL CHECK (state IN (
                             'RECORDED','REVERTING','REVERTED',
                             'PARTIAL_REVERT','HELD_FOR_HUMAN','FAILED_RETRYABLE'
                          )),
        lock_holder      TEXT,
        lock_expires_at  INTEGER,
        error            TEXT,
        created_at       INTEGER NOT NULL,
        updated_at       INTEGER NOT NULL
      );
    `);

    db.exec(`
      INSERT INTO lanes_new (
        id, task_id, sub_agent_id, checkpoint_id, endpoint, scenario,
        state, lock_holder, lock_expires_at, error, created_at, updated_at
      )
      SELECT
        id, task_id, sub_agent_id, checkpoint_id, endpoint, scenario,
        state, lock_holder, lock_expires_at, error, created_at, updated_at
      FROM lanes;
    `);

    db.exec(`DROP TABLE lanes;`);
    db.exec(`ALTER TABLE lanes_new RENAME TO lanes;`);

    // Re-create the full set of lane indexes from migration 001.
    db.exec(`
      CREATE INDEX idx_lanes_task_id       ON lanes(task_id);
      CREATE INDEX idx_lanes_sub_agent_id  ON lanes(sub_agent_id);
      CREATE INDEX idx_lanes_state         ON lanes(state);
      CREATE INDEX idx_lanes_checkpoint_id ON lanes(checkpoint_id);
      CREATE INDEX idx_lanes_endpoint      ON lanes(endpoint);
    `);

    // Correctness guard: foreign_key_check reports rows that violate FK
    // constraints in the current schema. If non-empty, throw to abort the
    // runner's wrapping transaction.
    const violations = db
      .prepare(`PRAGMA foreign_key_check('lanes')`)
      .all() as unknown[];
    if (violations.length > 0) {
      throw new Error(
        `migration 006 abort: foreign_key_check reported ${violations.length} violation(s) on lanes`,
      );
    }

    db.exec(`PRAGMA foreign_keys = ON;`);
  },
};
```

```ts
import type { Database } from 'better-sqlite3';
import { ulid } from 'ulid';
import type { TaskRow } from '../types.js';

/**
 * Per DESIGN_STORAGE.md §15 OQ-1 option C, we keep NULL task_id lanes from
 * the MCP-wedge period and expose a UI-facing helper that groups a selected
 * set into a freshly created task. Runs in a single transaction so partial
 * failure leaves neither a dangling task nor a half-reassigned set of lanes.
 */
export function assignLanesToNewTask(
  db: Database,
  laneIds: string[],
  input: { title: string },
): TaskRow {
  if (laneIds.length === 0) {
    throw new Error('assignLanesToNewTask: need at least one lane id');
  }
  const now = Date.now();
  const taskId = ulid();

  const createdRow = db.transaction(() => {
    // Verify every lane exists BEFORE inserting the task, so we fail fast
    // without leaving an orphan task row behind.
    const check = db.prepare(`SELECT 1 FROM lanes WHERE id = ?`);
    for (const id of laneIds) {
      const hit = check.get(id);
      if (!hit) {
        throw new Error(`assignLanesToNewTask: lane not found: ${id}`);
      }
    }

    db.prepare(
      `INSERT INTO tasks
         (id, title, description, description_vec, status, created_at, updated_at)
       VALUES (?, ?, NULL, NULL, 'PENDING', ?, ?)`,
    ).run(taskId, input.title, now, now);

    const upd = db.prepare(
      `UPDATE lanes SET task_id = ?, updated_at = ? WHERE id = ?`,
    );
    for (const id of laneIds) {
      upd.run(taskId, now, id);
    }

    return db
      .prepare(`SELECT * FROM tasks WHERE id = ?`)
      .get(taskId) as TaskRow;
  })();

  return createdRow;
}
```
