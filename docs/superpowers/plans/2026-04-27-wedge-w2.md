- [ ] **边界用例 1：中文 key**
  ```ts
  it('scratchpad 支持中文 key（"笔记:1"）', () => {
    const ws = openWorkspace({ cairnRoot });
    toolWriteScratch(ws, { key: '笔记:1', content: { text: '测试内容' } });
    const r = toolReadScratch(ws, { key: '笔记:1' });
    expect(r.found).toBe(true);
    expect(r.value).toEqual({ text: '测试内容' });
    ws.db.close();
  });
  ```

- [ ] **边界用例 2：超大 stash（200 KB+ blob 分流路径）**
  ```ts
  it('scratchpad 写入 200KB+ 内容走 blob 分流路径', () => {
    const ws = openWorkspace({ cairnRoot });
    const big = { data: 'x'.repeat(220_000) };
    toolWriteScratch(ws, { key: 'large-key', content: big });
    const r = toolReadScratch(ws, { key: 'large-key' });
    expect(r.found).toBe(true);
    expect((r.value as any).data).toHaveLength(220_000);
    ws.db.close();
  });
  ```

- [ ] **边界用例 3：非 git 仓库调 checkpoint.create**
  ```ts
  it('非 git 目录调 checkpoint.create 返回 stash_sha=null 而非 throw', () => {
    const tmpDir = mkdtempSync(join(tmpdir(), 'cairn-non-git-'));
    const ws = openWorkspace({ cairnRoot, cwd: tmpDir });
    let result: ReturnType<typeof toolCreateCheckpoint>;
    expect(() => {
      result = toolCreateCheckpoint(ws, { label: 'no-git' });
    }).not.toThrow();
    expect(result!.stash_sha).toBeNull();
    ws.db.close();
    rmSync(tmpDir, { recursive: true, force: true });
  });
  ```

- [ ] **边界用例 4：并发写同一个 scratchpad key**
  ```ts
  it('同一 key 两次写入后读取为最后一次值（last-write-wins）', () => {
    const ws = openWorkspace({ cairnRoot });
    toolWriteScratch(ws, { key: 'concurrent', content: 'first' });
    toolWriteScratch(ws, { key: 'concurrent', content: 'second' });
    const r = toolReadScratch(ws, { key: 'concurrent' });
    expect(r.value).toBe('second');
    ws.db.close();
  });
  ```

- [ ] **边界用例 5：rewind.preview 对不存在 checkpoint_id 的处理**

    expect(r).toHaveProperty('error');
    ws.db.close();
  });
