# Cairn Storage P2 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship the `checkpoints` table, two-phase commit capture pipeline, pluggable snapshot backends (rsync / APFS clonefile / git-stash), and startup recovery (CORRUPTED checkpoint scan + orphan blob GC) — so that by the time the desktop shell starts calling `captureCheckpoint`, the persistence layer can safely take file snapshots across crashes without leaking rows, directories, or blobs.

**Architecture:** Migration 002 adds the `checkpoints` table alongside the P1 schema. A new `checkpoints.ts` repository exposes five sync functions (create PENDING, mark READY, scan CORRUPTED, list, get). A two-phase commit helper `captureCheckpoint` splits into `db.tx` → file I/O → `db.tx` per DESIGN_STORAGE.md §7. Snapshot backends live in `src/storage/snapshots/` and implement a shared `SnapshotBackend` interface (`rsync` baseline, `apfs-clone` fast path on macOS, `git-stash` degraded fallback). Startup recovery (`runStartupRecovery`) composes the CORRUPTED scan with orphan blob GC — the daemon calls it once during boot before accepting writes.

**Tech Stack:** Same as P1 — Node.js 20+, TypeScript (strict), `better-sqlite3` v11+, `vitest`, `ulid`. Snapshot backends shell out to `rsync` / `cp -c -R` / `git` via `child_process.spawnSync` (sync, outside any `db.transaction`). No new npm deps.

**Spec:** [DESIGN_STORAGE.md](../../../DESIGN_STORAGE.md) — especially §4.4 (checkpoints DDL), §7 (transaction rules), §8 (two-phase commit + CORRUPTED scan), §15 OQ-2 (snapshot backend selection: rsync baseline + APFS clonefile fast path + git-stash degraded), §16 (P2 ships before desktop starts).

**Deviation from spec:** DESIGN_STORAGE.md §4.4 declares `checkpoints.task_id` as `REFERENCES tasks(id) ON DELETE SET NULL`. Because P2 does **not** create the `tasks` table (that is P3), migration 002 ships the column as `TEXT NULL` **without FK**; the FK will be re-established by P3's table-rewrite migration alongside `lanes.checkpoint_id`. This mirrors the equivalent P1 deviation on `lanes.task_id`.

```ts
import type { Database as DB } from 'better-sqlite3';
import { newId } from '../ids.js';
import type { CheckpointRow, CheckpointStatus } from '../types.js';

export interface NewCheckpoint {
  task_id?: string | null;
  label?: string | null;
  git_head?: string | null;
  snapshot_dir: string;
}

export function createPendingCheckpoint(db: DB, input: NewCheckpoint): CheckpointRow {
  const id = newId();
  const now = Date.now();
  db.prepare(
    `INSERT INTO checkpoints
       (id, task_id, label, git_head, snapshot_dir, snapshot_status,
        size_bytes, created_at, ready_at)
     VALUES (?, ?, ?, ?, ?, 'PENDING', NULL, ?, NULL)`
  ).run(
    id,
    input.task_id ?? null,
    input.label ?? null,
    input.git_head ?? null,
    input.snapshot_dir,
    now
  );
  return getCheckpointById(db, id)!;
}

export function markCheckpointReady(db: DB, id: string, sizeBytes: number): void {
  db.prepare(
    `UPDATE checkpoints
        SET snapshot_status = 'READY',
            size_bytes      = ?,
            ready_at        = ?
      WHERE id = ? AND snapshot_status = 'PENDING'`
  ).run(sizeBytes, Date.now(), id);
}

export function getCheckpointById(db: DB, id: string): CheckpointRow | null {
  const row = db
    .prepare('SELECT * FROM checkpoints WHERE id = ?')
    .get(id) as CheckpointRow | undefined;
  return row ?? null;
}

export function listCheckpointsByTask(
  db: DB,
  taskId: string | null,
  status?: CheckpointStatus
): CheckpointRow[] {
  if (status) {
    return db
      .prepare(
        `SELECT * FROM checkpoints
          WHERE task_id IS ? AND snapshot_status = ?
          ORDER BY created_at DESC`
      )
      .all(taskId, status) as CheckpointRow[];
  }
  return db
    .prepare(
      `SELECT * FROM checkpoints
        WHERE task_id IS ?
        ORDER BY created_at DESC`
    )
    .all(taskId) as CheckpointRow[];
}

export interface ScanCorruptedOpts {
  stalenessMs?: number;
  now?: number;
}

export function scanForCorruptedCheckpoints(
  db: DB,
  opts: ScanCorruptedOpts = {}
): CheckpointRow[] {
  const stalenessMs = opts.stalenessMs ?? 5 * 60 * 1000;
  const now = opts.now ?? Date.now();
  const cutoff = now - stalenessMs;

  const stale = db
    .prepare(
      `SELECT * FROM checkpoints
        WHERE snapshot_status = 'PENDING' AND created_at < ?`
    )
    .all(cutoff) as CheckpointRow[];

  if (stale.length === 0) return [];

  const update = db.prepare(
    `UPDATE checkpoints
        SET snapshot_status = 'CORRUPTED'
      WHERE id = ? AND snapshot_status = 'PENDING'`
  );
  const tx = db.transaction((rows: CheckpointRow[]) => {
    for (const r of rows) update.run(r.id);
  });
  tx(stale);

  return stale.map((r) => ({ ...r, snapshot_status: 'CORRUPTED' as const }));
}
```

```ts
  it('ignores fresh PENDING rows inside staleness window', () => {
    createPendingCheckpoint(db, { snapshot_dir: '/tmp/fresh' });
    const flipped = scanForCorruptedCheckpoints(db, { stalenessMs: 5 * 60 * 1000 });
```

```ts
import { spawnSync } from 'node:child_process';
import { mkdirSync } from 'node:fs';
import { dirname } from 'node:path';

export interface RsyncOpts {

}

export function isRsyncAvailable(): boolean {
  const r = spawnSync('rsync', ['--version'], { stdio: 'ignore' });
  return r.status === 0;
}

export function rsyncSnapshot(sourceDir: string, targetDir: string, opts: RsyncOpts = {}): void {
  if (!isRsyncAvailable()) {
    throw new Error(
      'rsync is required for snapshots but was not found on PATH. ' +
        'Install it with: macOS `brew install rsync`, ' +
        'Debian/Ubuntu `sudo apt-get install rsync`, ' +
        'Windows `choco install rsync` or use WSL.'
    );
  }

  mkdirSync(dirname(targetDir), { recursive: true });
  mkdirSync(targetDir, { recursive: true });

  const args: string[] = ['-a'];
  if (opts.linkDestDir) args.push('--link-dest', opts.linkDestDir);
  // Trailing slashes: copy *contents* of source into target.
  args.push(`${sourceDir.replace(/[/\\]+$/, '')}/`, `${targetDir.replace(/[/\\]+$/, '')}/`);

  const result = spawnSync('rsync', args, { stdio: ['ignore', 'pipe', 'pipe'] });
  if (result.status !== 0) {
    const stderr = result.stderr?.toString() ?? '';
    throw new Error(
      `rsync failed with exit code ${result.status}: ${stderr.trim() || 'unknown error'}`
    );
  }
}
```

```ts
import { spawnSync } from 'node:child_process';
import { mkdirSync } from 'node:fs';
import { dirname } from 'node:path';
import { rsyncSnapshot } from './rsync.js';

export function isApfsCloneSupported(): boolean {
  return process.platform === 'darwin';
}

/**
 * Fast-path snapshot on macOS using APFS clonefile (copy-on-write) via `cp -c -R`.
 * Falls back to rsync if `cp -c` is unavailable or the filesystem rejects the flag
 * (e.g. non-APFS volumes, exFAT externals).
 */
export function apfsCloneSnapshot(sourceDir: string, targetDir: string): void {
  if (!isApfsCloneSupported()) {
    // Degrade silently to rsync on non-darwin; pickSnapshotBackend normally prevents this.
    rsyncSnapshot(sourceDir, targetDir);
    return;
  }

  mkdirSync(dirname(targetDir), { recursive: true });

  // Note: BSD cp requires the target to NOT exist for recursive copies to work
  // cleanly, so we do not pre-create targetDir here.
  const result = spawnSync(
    'cp',
    ['-c', '-R', `${sourceDir.replace(/[/\\]+$/, '')}/.`, targetDir],
    { stdio: ['ignore', 'pipe', 'pipe'] }
  );

  if (result.status === 0) return;

  // Fallback: cp -c failed (unsupported FS, old cp). Use rsync.
  rsyncSnapshot(sourceDir, targetDir);
}
```

```ts
import { spawnSync } from 'node:child_process';
import { existsSync, mkdirSync, writeFileSync } from 'node:fs';
import { join } from 'node:path';

export interface GitStashResult {
  /** Resulting stash commit SHA, or null if there was nothing to stash. */
  sha: string | null;
}

export function isGitRepo(dir: string): boolean {
  if (!existsSync(join(dir, '.git'))) {
    const r = spawnSync('git', ['-C', dir, 'rev-parse', '--is-inside-work-tree'], {
      stdio: ['ignore', 'pipe', 'pipe'],
    });
    return r.status === 0 && r.stdout.toString().trim() === 'true';
  }
  return true;
}

/**
 * Degraded snapshot: captures tracked changes only via `git stash create`.
 *
 * Per DESIGN_STORAGE.md §15 OQ-2, this is the worst-case fallback (see
 * PRODUCT.md §11.2). Untracked files are NOT captured. The returned SHA is
 * an unreachable commit object in the source repository's ODB; the caller
 * is responsible for arranging that it is not garbage-collected (we rely
 * on git's default 90-day reflog window plus the explicit rewind workflow
 * bumping the reflog before `git gc` runs).
 */
export function gitStashSnapshot(sourceDir: string, targetDir: string): GitStashResult {
  if (!isGitRepo(sourceDir)) {
    throw new Error(`gitStashSnapshot requires a git repo, got: ${sourceDir}`);
  }

  mkdirSync(targetDir, { recursive: true });

  const r = spawnSync('git', ['-C', sourceDir, 'stash', 'create'], {
    stdio: ['ignore', 'pipe', 'pipe'],
  });
  if (r.status !== 0) {
    const stderr = r.stderr?.toString() ?? '';
    throw new Error(`git stash create failed: ${stderr.trim() || 'unknown'}`);
  }
  const sha = r.stdout.toString().trim() || null;

  writeFileSync(join(targetDir, 'stash-sha.txt'), (sha ?? '') + '\n');
  writeFileSync(
    join(targetDir, 'README.txt'),
    'Degraded checkpoint: git stash SHA recorded in stash-sha.txt.\n' +
      'Untracked files were NOT captured. See PRODUCT.md §11.2.\n'
  );

  return { sha };
}
```

```ts
import { apfsCloneSnapshot, isApfsCloneSupported } from './apfs-clone.js';
import { gitStashSnapshot } from './git-stash.js';

export type SnapshotBackendName = 'apfs-clone' | 'rsync' | 'git-stash';

export interface SnapshotBackend {
  name: SnapshotBackendName;
  /** Materialize sourceDir contents into targetDir. Throws on failure. */
  snapshot: (sourceDir: string, targetDir: string) => void;
}

export interface PickOpts {
  /** Force the degraded git-stash fallback (used when rsync + clonefile are unsafe). */
  preferGitStash?: boolean;
}

export function pickSnapshotBackend(opts: PickOpts = {}): SnapshotBackend {
  if (opts.preferGitStash) {
    return {
      name: 'git-stash',
      snapshot: (s, t) => {
        gitStashSnapshot(s, t);
      },
    };
  }
  if (isApfsCloneSupported()) {
    return {
      name: 'apfs-clone',
      snapshot: apfsCloneSnapshot,
    };
  }
  if (isRsyncAvailable()) {
    return {
      name: 'rsync',
      snapshot: rsyncSnapshot,
    };
  }
  // Last resort: git-stash (may still fail if source isn't a git repo, but the
  // caller gets a clear error from gitStashSnapshot at capture time).
  return {
    name: 'git-stash',
    snapshot: (s, t) => {
      gitStashSnapshot(s, t);
    },
  };
}

export { rsyncSnapshot, apfsCloneSnapshot, gitStashSnapshot };
```

```ts
import { spawnSync } from 'node:child_process';
import { existsSync, mkdirSync, readdirSync, statSync } from 'node:fs';
import { join } from 'node:path';
import type { Database as DB } from 'better-sqlite3';
import type { CheckpointRow } from './types.js';
import {
  createPendingCheckpoint,
  markCheckpointReady,
  getCheckpointById,
} from './repositories/checkpoints.js';
import { pickSnapshotBackend, type SnapshotBackend } from './snapshots/index.js';

export interface CaptureOpts {
  /** Absolute path of the directory whose contents should be snapshotted. */
  sourceDir: string;
  task_id?: string | null;
  label?: string | null;
  /** Inject a specific backend; defaults to pickSnapshotBackend(). */
  backend?: SnapshotBackend;
  /** Override to force the degraded git-stash path. */
  preferGitStash?: boolean;
}

function captureGitHead(sourceDir: string): string | null {
  const r = spawnSync('git', ['-C', sourceDir, 'rev-parse', 'HEAD'], {
    stdio: ['ignore', 'pipe', 'pipe'],
  });
  if (r.status !== 0) return null;
  const out = r.stdout.toString().trim();
  return out || null;
}

/** Recursively sum byte sizes under `dir`. Mirrors `du -sb`. */
function duBytes(dir: string): number {
  if (!existsSync(dir)) return 0;
  let total = 0;
  for (const entry of readdirSync(dir, { withFileTypes: true })) {
    const p = join(dir, entry.name);
    if (entry.isDirectory()) total += duBytes(p);
    else if (entry.isFile()) total += statSync(p).size;
  }
  return total;
}

/**
 * Two-phase commit checkpoint capture (DESIGN_STORAGE.md §8).
 *
 *   ① db.tx:  INSERT checkpoints(snapshot_status='PENDING', snapshot_dir=...)
 *   ② fs:     backend.snapshot(sourceDir -> snapshot_dir)      (NO db.tx held)
 *   ③ db.tx:  UPDATE snapshot_status='READY', size_bytes, ready_at=now
 *
 * Any failure between ① and ③ leaves the row as PENDING; the startup
 * `scanForCorruptedCheckpoints` will reap it.
 */
export function captureCheckpoint(
  db: DB,
  snapshotRoot: string,
  opts: CaptureOpts
): CheckpointRow {
  const backend = opts.backend ?? pickSnapshotBackend({ preferGitStash: opts.preferGitStash ?? false });

  mkdirSync(snapshotRoot, { recursive: true });

  // -------- Phase 1: reserve a row and a unique directory path --------
  // We cannot know the id before INSERT (it is generated in the repo), so we
  // do the insert first, then use the returned id to derive snapshot_dir.
  // To keep the insert inside a single db.tx without I/O, we use a placeholder
  // snapshot_dir then UPDATE to the real path in the same tx-less stretch.
  const placeholder = join(snapshotRoot, 'pending-' + Date.now() + '-' + Math.random().toString(36).slice(2, 10));
  const pending = createPendingCheckpoint(db, {
    task_id: opts.task_id ?? null,
    label: opts.label ?? null,
    snapshot_dir: placeholder,
  });

  const finalDir = join(snapshotRoot, pending.id);
  db.prepare('UPDATE checkpoints SET snapshot_dir = ? WHERE id = ?').run(finalDir, pending.id);

  // -------- Phase 2: file I/O outside any db.transaction --------
  const gitHead = captureGitHead(opts.sourceDir);
  if (gitHead) {
    db.prepare('UPDATE checkpoints SET git_head = ? WHERE id = ?').run(gitHead, pending.id);
  }
  backend.snapshot(opts.sourceDir, finalDir);

  const size = duBytes(finalDir);

  // -------- Phase 3: flip to READY --------
  markCheckpointReady(db, pending.id, size);

  const final = getCheckpointById(db, pending.id);
  if (!final) throw new Error(`checkpoint ${pending.id} vanished after markReady`);
  return final;
}
```

```ts
import { readdirSync, rmSync, statSync, existsSync, unlinkSync } from 'node:fs';
import { join } from 'node:path';
import type { Database as DB } from 'better-sqlite3';
import { scanForCorruptedCheckpoints } from './repositories/checkpoints.js';

export interface RecoveryReport {
  corruptedCheckpoints: number;
  orphanBlobsDeleted: number;
}

/**
 * Walk blobRoot recursively and return absolute paths of every regular file.
 * Missing root returns an empty list.
 */
function listBlobFiles(blobRoot: string): string[] {
  if (!existsSync(blobRoot)) return [];
  const out: string[] = [];
  const stack: string[] = [blobRoot];
  while (stack.length) {
    const dir = stack.pop()!;
    let entries;
    try {
      entries = readdirSync(dir, { withFileTypes: true });
    } catch {
      continue;
    }
    for (const e of entries) {
      const p = join(dir, e.name);
      if (e.isDirectory()) stack.push(p);
      else if (e.isFile()) out.push(p);
    }
  }
  return out;
}

/**
 * Collect every blob-path reference from the DB. Currently queries ops +
 * compensations columns; extend here when new tables grow *_path columns.
 */
function referencedBlobPaths(db: DB): Set<string> {
  const refs = new Set<string>();
  const cols = [
    ['ops',           'request_body_path'],
    ['ops',           'response_body_path'],
    ['ops',           'before_image_path'],
    ['compensations', 'payload_path'],
  ] as const;

  const tables = new Set(
    (db
      .prepare("SELECT name FROM sqlite_master WHERE type='table'")
      .all() as { name: string }[]).map((r) => r.name)
  );

  for (const [table, col] of cols) {
    if (!tables.has(table)) continue;
    const rows = db
      .prepare(`SELECT ${col} AS p FROM ${table} WHERE ${col} IS NOT NULL`)
      .all() as { p: string }[];
    for (const r of rows) refs.add(r.p);
  }
  return refs;
}

const ORPHAN_GRACE_MS = 10 * 60 * 1000;

export function scanOrphanBlobs(db: DB, blobRoot: string): number {
  const all = listBlobFiles(blobRoot);
  if (all.length === 0) return 0;
  const refs = referencedBlobPaths(db);
  const cutoff = Date.now() - ORPHAN_GRACE_MS;

  let deleted = 0;
  for (const p of all) {
    if (refs.has(p)) continue;
    let mtime;
    try {
      mtime = statSync(p).mtimeMs;
    } catch {
      continue;
    }
    if (mtime >= cutoff) continue; // too fresh — may be mid-write
    try {
      unlinkSync(p);
      deleted++;
    } catch {
      // swallow: file may have been raced away; not fatal for GC
    }
  }
  return deleted;
}

/**
 * Daemon boot recovery: reap stale PENDING checkpoints, delete their
 * snapshot directories, then GC orphan blobs. Safe to call repeatedly.
 */
export function runStartupRecovery(db: DB, blobRoot: string): RecoveryReport {
  const flipped = scanForCorruptedCheckpoints(db);
  for (const row of flipped) {
    try {
      rmSync(row.snapshot_dir, { recursive: true, force: true });
    } catch {
      // swallow: best-effort cleanup; DB row already marks it CORRUPTED.
    }
  }

  const orphanBlobsDeleted = scanOrphanBlobs(db, blobRoot);

  return {
    corruptedCheckpoints: flipped.length,
    orphanBlobsDeleted,
  };
}
```
