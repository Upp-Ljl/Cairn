```ts
import type { Database as DB } from 'better-sqlite3';
import { writeBlobIfLarge, readBlob } from '../blobs.js';
import type { ScratchpadRow } from '../types.js';

export interface PutScratchInput {
  key: string;
  value: unknown;
  task_id?: string | null;
  expires_at?: number | null;
}

export function putScratch(db: DB, blobRoot: string, input: PutScratchInput): void {
  const now = Date.now();
  const ref = writeBlobIfLarge(input.value, blobRoot);
  db.prepare(`
    INSERT INTO scratchpad
      (key, value_json, value_path, task_id, expires_at, created_at, updated_at)
    VALUES (@key, @value_json, @value_path, @task_id, @expires_at, @created_at, @updated_at)
    ON CONFLICT(key) DO UPDATE SET
      value_json = excluded.value_json,
      value_path = excluded.value_path,
      task_id    = excluded.task_id,
      expires_at = excluded.expires_at,
      updated_at = excluded.updated_at
  `).run({
    key: input.key,
    value_json: ref.json ?? null,
    value_path: ref.path ?? null,
    task_id: input.task_id ?? null,
    expires_at: input.expires_at ?? null,
    created_at: now,
    updated_at: now,
  });
}

export function getScratch(db: DB, key: string): unknown | null {
  const row = db.prepare(
    'SELECT value_json, value_path FROM scratchpad WHERE key = ?'
  ).get(key) as { value_json: string | null; value_path: string | null } | undefined;
  if (!row) return null;
  return readBlob({
    ...(row.value_json !== null ? { json: row.value_json } : {}),
    ...(row.value_path !== null ? { path: row.value_path } : {}),
  });
}

export function listAllScratch(db: DB): ScratchpadRow[] {
  return db.prepare(
    'SELECT * FROM scratchpad ORDER BY updated_at DESC'
  ).all() as ScratchpadRow[];
}

export function deleteScratch(db: DB, key: string): void {
  db.prepare('DELETE FROM scratchpad WHERE key = ?').run(key);
}
```

- [ ] **Step 1**: 写测试。Git-stash 后端的契约：给定一个 git 仓库目录 + 一个 snapshot id，返回一个字符串（stash 提交 SHA），后续可用于还原。

```ts
import { describe, it, expect } from 'vitest';
import { execSync } from 'node:child_process';
import { mkdtempSync, writeFileSync, readFileSync, rmSync } from 'node:fs';
import { join } from 'node:path';
import { tmpdir } from 'node:os';
import { gitStashSnapshot, gitStashRestore } from '../../src/storage/snapshots/git-stash.js';

function makeRepo() {
  const dir = mkdtempSync(join(tmpdir(), 'cairn-git-'));
  execSync('git init -q', { cwd: dir });
  execSync('git config user.email test@cairn.local', { cwd: dir });
  execSync('git config user.name Test', { cwd: dir });
  writeFileSync(join(dir, 'a.txt'), 'original');
  execSync('git add . && git commit -q -m init', { cwd: dir });
  return dir;
}

describe('gitStashSnapshot + gitStashRestore', () => {
  it('captures dirty changes and restores them', () => {
    const repo = makeRepo();
    try {
      writeFileSync(join(repo, 'a.txt'), 'modified');
      writeFileSync(join(repo, 'b.txt'), 'new');
      execSync('git add b.txt', { cwd: repo }); // tracked but uncommitted

      const sha = gitStashSnapshot(repo);
      expect(sha).toMatch(/^[0-9a-f]{40}$/);

      // 之后用户继续改文件
      writeFileSync(join(repo, 'a.txt'), 'further changes');

      // 还原
      gitStashRestore(repo, sha);
      expect(readFileSync(join(repo, 'a.txt'), 'utf8')).toBe('modified');
      expect(readFileSync(join(repo, 'b.txt'), 'utf8')).toBe('new');
    } finally {
      rmSync(repo, { recursive: true, force: true });
    }
  });

  it('returns null SHA when working tree is clean', () => {
    const repo = makeRepo();
    try {
      expect(gitStashSnapshot(repo)).toBeNull();
    } finally {
      rmSync(repo, { recursive: true, force: true });
    }
  });
```

```ts
import { execSync } from 'node:child_process';

/**
 * 在 cwd 上创建一个 stash 对象（不进 stash 栈），返回 stash commit SHA。
 * 工作树干净时返回 null。
 * 不修改工作树（git stash create 只创建对象，不动 working tree）。
 */
export function gitStashSnapshot(cwd: string): string | null {
  // 添加未跟踪文件到 index 临时（避免 stash 漏掉）
  execSync('git add -A', { cwd });
  const sha = execSync('git stash create', { cwd, encoding: 'utf8' }).trim();
  // 还原 staging 状态（reset 仅 index，不动 working tree）
  execSync('git reset', { cwd });
  return sha === '' ? null : sha;
}

/**
 * 把工作树还原到指定 stash SHA 的状态。
 * 用 `git checkout -- <files>` 路径而非 `git stash apply`，避免污染 stash 栈。
 * 楔期约定：不动 .git/HEAD（PRODUCT.md §17 楔期 rewind 约定）。
 */
export function gitStashRestore(cwd: string, stashSha: string): void {
  // 列出 stash 涉及的文件
  const files = execSync(
    `git diff-tree --no-commit-id --name-only -r ${stashSha}`,
    { cwd, encoding: 'utf8' }
  ).trim().split('\n').filter(Boolean);
  if (files.length === 0) return;
  // 用 git restore --source=<sha> 还原每个文件到 stash 时刻
  execSync(`git restore --source=${stashSha} -- ${files.map((f) => `"${f}"`).join(' ')}`, { cwd });
}

/**
 * 给定 stash SHA，返回会被还原影响的文件名清单（rewind preview 用）。
 */
export function gitStashAffectedFiles(cwd: string, stashSha: string): string[] {
  return execSync(
    `git diff-tree --no-commit-id --name-only -r ${stashSha}`,
    { cwd, encoding: 'utf8' }
  ).trim().split('\n').filter(Boolean);
}
```

```ts
import type { Database as DB } from 'better-sqlite3';
import { newId } from '../ids.js';
import type { CheckpointRow, CheckpointStatus } from '../types.js';

export interface NewCheckpoint {
  task_id?: string | null;
  label?: string | null;
  snapshot_dir: string;
}

export function createPendingCheckpoint(db: DB, input: NewCheckpoint): CheckpointRow {
  const now = Date.now();
  const row: CheckpointRow = {
    id: newId(),
    task_id: input.task_id ?? null,
    label: input.label ?? null,
    git_head: null,
    snapshot_dir: input.snapshot_dir,
    snapshot_status: 'PENDING',
    size_bytes: null,
    created_at: now,
    ready_at: null,
  };
  db.prepare(`
    INSERT INTO checkpoints
      (id, task_id, label, git_head, snapshot_dir, snapshot_status,
       size_bytes, created_at, ready_at)
    VALUES (@id, @task_id, @label, @git_head, @snapshot_dir, @snapshot_status,
            @size_bytes, @created_at, @ready_at)
  `).run(row);
  return row;
}

export function markCheckpointReady(
  db: DB,
  id: string,
  attrs: { size_bytes: number; git_head: string | null }
): void {
  db.prepare(`
    UPDATE checkpoints
       SET snapshot_status = 'READY',
           size_bytes = ?,
           git_head = ?,
           ready_at = ?
     WHERE id = ? AND snapshot_status = 'PENDING'
  `).run(attrs.size_bytes, attrs.git_head, Date.now(), id);
}

export function getCheckpointById(db: DB, id: string): CheckpointRow | null {
  const row = db.prepare('SELECT * FROM checkpoints WHERE id = ?').get(id) as CheckpointRow | undefined;
  return row ?? null;
}

export function listCheckpoints(db: DB, status?: CheckpointStatus): CheckpointRow[] {
  if (status) {
    return db.prepare(
      'SELECT * FROM checkpoints WHERE snapshot_status = ? ORDER BY created_at DESC'
    ).all(status) as CheckpointRow[];
  }
  return db.prepare(
    'SELECT * FROM checkpoints ORDER BY created_at DESC'
  ).all() as CheckpointRow[];
}
```

```ts
import type { Workspace } from '../workspace.js';
import { getCheckpointById } from '@cairn/daemon/src/storage/repositories/checkpoints.js';
import {
  gitStashRestore, gitStashAffectedFiles,
} from '@cairn/daemon/src/storage/snapshots/git-stash.js';

function extractStashSha(label: string | null): string | null {
  if (!label) return null;
  const m = label.match(/::stash:([0-9a-f]{40})/);
  return m ? m[1]! : null;
}

export function toolRewindPreview(ws: Workspace, args: { checkpoint_id: string }) {
  const c = getCheckpointById(ws.db, args.checkpoint_id);
  if (!c) return { error: 'checkpoint not found' };
  const sha = extractStashSha(c.label);
  if (!sha) return { error: 'no stash backend recorded', files: [] };
  const files = gitStashAffectedFiles(ws.cwd, sha);
  return { checkpoint_id: c.id, files, git_head_at_checkpoint: c.git_head };
}

export function toolRewindTo(ws: Workspace, args: { checkpoint_id: string }) {
  const c = getCheckpointById(ws.db, args.checkpoint_id);
  if (!c) return { ok: false, error: 'checkpoint not found' };
  const sha = extractStashSha(c.label);
  if (!sha) return { ok: false, error: 'no stash backend recorded' };
  gitStashRestore(ws.cwd, sha);
  return { ok: true, restored_files: gitStashAffectedFiles(ws.cwd, sha) };
}
```
